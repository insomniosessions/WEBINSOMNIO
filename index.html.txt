<script>

const phone = "56912345678";

const customTicketPrice = 10000;

const products = [

        price:10000,
        subtitle:"Hasta 19 jun 2026"
    },

        icon:"🎟️",
        title:"GRUPO 5 PERSONAS | 5 ENTRADAS",
        price:45000,
        subtitle:"Pack especial"
    },
    {
        id:"p4",
        icon:"🎟️",
        title:"GRUPO 5 PERSONAS | 5 ENTRADAS + 10 COVER",
        price:55000,
        subtitle:"Pack especial"
    },
    {
        id:"p5",
        icon:"🎟️",
        title:"GRUPO 10 PERSONAS | 10 ENTRADAS",
        price:80000,
        subtitle:"Pack especial"
    },
    {
        id:"p6",
        icon:"🎟️",
        title:"GRUPO 10 PERSONAS | 10 ENTRADAS + 10 SHOT TEQUILA + 10 COVER",
        price:100000,
        subtitle:"Pack especial"
    }
];

const state = {};

products.forEach(p => state[p.id] = 0);

function formatCLP(n){
    return n.toLocaleString("es-CL");
}

function renderOptions(){

    document.getElementById("options").innerHTML =
    products.map(p => `
    <div class="option-card">

        <div class="option-left">

            <div class="icon-box">
                ${p.icon}
            </div>

            <div>
                <div class="option-title">
                    ${p.title}
                </div>

                <div class="option-price">
                    $ ${formatCLP(p.price)}
                </div>

                <div class="option-sub">
                    ${p.subtitle}
                </div>
            </div>

        </div>

        <div class="option-controls">

            <button
                class="qty-btn"
                onclick="changeQty('${p.id}',-1)">
                −
            </button>

            <div
                class="qty-box"
                id="qty-${p.id}">
                0
            </div>

            <button
                class="qty-btn"
                onclick="changeQty('${p.id}',1)">
                +
            </button>

        </div>

    </div>
    `).join("");
}

function changeQty(id, delta){

    state[id] = Math.max(
        0,
        state[id] + delta
    );

    document.getElementById(
        `qty-${id}`
    ).innerText = state[id];

    updateSummary();
}

function updateSummary(){

    let total = 0;
    let items = 0;
    let lines = [];

    products.forEach(p=>{

        const qty = state[p.id];

        if(qty > 0){

            const subtotal =
                qty * p.price;

            total += subtotal;
            items += qty;

            lines.push(
                `• ${p.title} x${qty} = $${formatCLP(subtotal)}`
            );
        }
    });

    const customQty =
        parseInt(
            document.getElementById(
                "customQty"
            ).value || 0
        );

    if(customQty > 0){

        const customTotal =
            customQty *
            customTicketPrice;

        total += customTotal;
        items += customQty;

        lines.push(
            `• TICKETS PERSONALIZADOS x${customQty} = $${formatCLP(customTotal)}`
        );
    }

    document.getElementById(
        "totalItems"
    ).innerText = items;

    document.getElementById(
        "totalPrice"
    ).innerText = formatCLP(total);

    document.getElementById(
        "purchaseSummary"
    ).innerHTML =
        lines.length
        ? lines.join("<br>")
        : "No hay tickets seleccionados.";

    const message =
        lines.length
        ? `Hola 👋

Quiero comprar:

${lines.join("\n")}

Total a pagar: $${formatCLP(total)}`
        : `Hola 👋

Quiero comprar entradas.`;

    document.getElementById(
        "whatsappBtn"
    ).href =
    `https://wa.me/${phone}?text=${encodeURIComponent(message)}`;
}

renderOptions();
updateSummary();
